% ---- PARTS (DIO I–V): eyebrow + title + verdigris rule --------------

% Framed part divider (ochre rule above, verdigris below) — rhymes with booklet.
\newcommand{\StatecraftPartHead}[1]{%
  \raggedright
  {\color{ochre}\rule{\linewidth}{0.8pt}}\par
  \vskip 0.85em
  {\nrsemibold\color{ink}\fontsize{28}{33}\selectfont #1\par}%
  \vskip 0.85em
  {\color{verdigris}\rule{\linewidth}{1.5pt}}%
}
\renewcommand{\partlineswithprefixformat}[3]{\StatecraftPartHead{#3}}

% ---- CHAPTERS: eyebrow + title + verdigris rule + chapter-front air --
% Eyebrow says "Poglavlje" for body chapters; \appendix turns it off so the
% bibliography ("Literatura") is not mislabelled.
\newif\ifsceyebrow \sceyebrowtrue
\let\scoldappendix\appendix
\renewcommand{\appendix}{\scoldappendix\sceyebrowfalse}
\setkomafont{chapter}{\nrsemibold\color{ink}}
\renewcommand*{\chapterformat}{}
\renewcommand*{\chapterheadstartvskip}{\null\vskip 5.5em}
\newcommand{\StatecraftChapHead}[1]{%
  \raggedright
  \ifsceyebrow {\eyebrow{Poglavlje}\par}\vskip 0.55em \fi
  {\usekomafont{chapter}\fontsize{22}{26}\selectfont #1\par}%
  \vskip 0.45em
  {\color{verdigris}\rule{\linewidth}{1.3pt}}%
  \par\vskip 2.2em
}
\renewcommand{\chapterlinesformat}[3]{\StatecraftChapHead{#3}}
\renewcommand{\chapterlineswithprefixformat}[3]{\StatecraftChapHead{#3}}

% =====================================================================
%  THEOREM NUMBERING — flat & continuous (sections are unnumbered, so the
%  default "Definicija 0.0.1" is meaningless). remreset is installed; we drop
%  the per-section reset and number each kind continuously across the book.
% =====================================================================
\usepackage{remreset}
\makeatletter
\newcommand{\sc@flatten}[1]{%
  \@ifundefined{c@#1}{}{\@removefromreset{#1}{section}\@namedef{the#1}{\arabic{#1}}}}
\AtBeginDocument{%
  \sc@flatten{definition}\sc@flatten{proposition}\sc@flatten{theorem}%
  \sc@flatten{lemma}\sc@flatten{corollary}\sc@flatten{example}\sc@flatten{exercise}%
}
\makeatother
